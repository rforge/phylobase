00028 // splits a string by whitespace and push the graphical strings to the back of r.
00029 //  Leading and trailing whitespace is lost ( there will be no empty strings added
00030 //      to the list.
00031 void NxsString::split(const std::string &s, std::list<std::string> * r)
00032     {
00033     NCL_ASSERT(r);
00034     if (r == NULL)
00035         return;
00036     std::string current;
00037     for (std::string::const_iterator sIt = s.begin(); sIt != s.end(); ++sIt)
00038         {
00039         const char c = *sIt;
00040         if (isgraph(c))
00041             current.append(1, c);
00042         else if (!current.empty())
00043             {
00044             r->push_back(current);
00045             current.clear();
00046             }
00047         }
00048     if (!current.empty())
00049         r->push_back(current);
00050     }

00259 int NxsString::PrintF(
00260   const char *formatStr,    /* the printf-style format string */
00261   ...)                      /* other arguments referred to by the format string */
00262     {
00263     const int kInitialBufferSize = 256;
00264     char buf[kInitialBufferSize];
00265 
00266     // Create a pointer to the list of optional arguments
00267     //
00268     va_list argList;
00269 
00270     // Set arg_ptr to the first optional argument in argList. The
00271     // second argument (formatStr) is the last non-optional argument.
00272     //
00273     va_start(argList, formatStr);
00274 
00275     // If vsnprintf returns -1, means kInitialBufferSize was not large enough.
00276     // In this case, only kInitialBufferSize bytes are written.
00277     //
00278     int nAdded = vsnprintf(buf, kInitialBufferSize, formatStr, argList);
00279 
00280     // Reset the argument list pointer
00281     //
00282     va_end(argList);
00283 
00284     // Currently, if formatted string is too long to fit into the supplied buf,
00285     // just adding a terminating '\0' and returning the truncated string
00286     // Need to think of a better solution
00287     //
00288     if (nAdded  < 0 || nAdded >= kInitialBufferSize)
00289         buf[kInitialBufferSize - 1] = '\0';
00290 
00291     *this << buf;
00292 
00293     return nAdded;
00294     }

00299 bool NxsString::IsStdAbbreviation(
00300   const NxsString &s,   /* the string for which the stored string is potentially an abbreviation */
00301   bool respectCase)     /* if true, comparison will be case-sensitive */
00302   const
00303     {
00304     if (empty())
00305         return false;
00306 
00307     // s is the unabbreviated comparison string
00308     //
00309     const unsigned slen = static_cast<unsigned long>(s.size());
00310 
00311     // t is the stored string
00312     //
00313     const unsigned tlen = static_cast<unsigned long>(size());
00314 
00315     // t cannot be an abbreviation of s if it is longer than s
00316     //
00317     if (tlen > slen)
00318         return false;
00319 
00320     // Examine each character in t and return false (meaning "not an abbreviation")
00321     // if at any point the corresponding character in s is different
00322     //
00323     for (unsigned k = 0; k < tlen; k++)
00324         {
00325         if (respectCase)
00326             {
00327             if ((*this)[k] != s[k])
00328                 return false;
00329             }
00330         else if (toupper((*this)[k]) != toupper(s[k]))
00331             return false;
00332         }
00333 
00334     return true;
00335     }

00343 bool NxsString::IsCapAbbreviation(
00344   const NxsString &s)   /* the string for which the stored string is potentially an abbreviation */
00345   const
00346     {
00347     if (empty())
00348         return false;
00349 
00350     // s is the unabbreviated comparison string
00351     //
00352     const unsigned slen = static_cast<unsigned>(s.size());
00353 
00354     // t is the stored string
00355     //
00356     const unsigned tlen = static_cast<unsigned>(size());
00357 
00358     // If the stored string is longer than s then it cannot be an abbreviation of s
00359     //
00360     if (tlen > slen)
00361         return false;
00362 
00363     unsigned k = 0;
00364     for (; k < slen; k++)
00365         {
00366         if (isupper(s[k]))
00367             {
00368             // If still in the uppercase portion of s and we've run out of characters
00369             // in t, then t is not a valid abbrevation of s
00370             //
00371             if (k >= tlen)
00372                 return false;
00373 
00374             // If kth character in t is not equal to kth character in s, then
00375             // t is not an abbrevation of s
00376             //
00377             char tokenChar = (char)toupper((*this)[k]);
00378             if (tokenChar != s[k])
00379                 return false;
00380             }
00381         else if (!isalpha(s[k]))
00382             {
00383             // Get here if we are no longer in the upper case portion of s and
00384             // s[k] is not an alphabetic character. This section is necessary because
00385             // we are dealing with a section of s that is not alphabetical and thus
00386             // we cannot tell whether this should be part of the abbrevation or not
00387             // (i.e. we cannot tell if it is capitalized or not). In this case, we
00388             // pretend that we are still in the upper case portion of s and return
00389             // false if we have run out of characters in t (meaning that the abbreviation
00390             // was too short) or we find a mismatch.
00391             //
00392             if (k >= tlen)
00393                 return false;
00394 
00395             if ((*this)[k] != s[k])
00396                 return false;
00397             }
00398         else
00399             {
00400             // Get here if we are no longer in the upper case portion of s and
00401             // s[k] is an alphabetic character. Just break because we have determined
00402             // that t is in fact a valid abbreviation of s.
00403             //
00404             break;
00405             }
00406         }
00407 
00408     // Check the lower case portion of s and any corresponding characters in t for mismatches
00409     // Even though the abbreviation is valid up to this point, it will become invalid if
00410     // any mismatches are found beyond the upper case portion of s
00411     //
00412     for (; k < tlen; k++)
00413         {
00414         const char tokenChar = (char)toupper((*this)[k]);
00415         const char otherChar = (char)toupper(s[k]);
00416         if (tokenChar != otherChar)
00417             return false;
00418         }
00419 
00420     return true;
00421     }

00428 NxsString &NxsString::RightJustifyLong(
00429   long x,           /* long value to right justify */
00430   unsigned int w,   /* width of field */
00431   bool clear_first) /* if true, initialize string first to empty string */
00432     {
00433     bool x_negative = (x < 0L ? true : false);
00434     unsigned long xabs = (x_negative ? (-x) : x);
00435     unsigned num_spaces = w;
00436 
00437     // If w = 10 and x = 123, we need 7 blank spaces before x
00438     // log10(123) is 2.09, indicating that x is at least 10^2 = 100 but not
00439     // 10^3 = 1000, thus x requires at least 3 characters to display
00440     //
00441     unsigned x_width = (x == 0 ? 1 :1 + (int)log10((double)xabs));
00442     if (x_negative)
00443         x_width++;  // for the minus sign
00444 
00445     NCL_ASSERT(x_width <= num_spaces);
00446     num_spaces -= x_width;
00447 
00448     if (clear_first)
00449         erase();
00450 
00451     for (unsigned k = 0; k < num_spaces; k++)
00452         *this += ' ';
00453 
00454     if (x_negative)
00455         *this += '-';
00456 
00457     *this += xabs;
00458     return *this;
00459     }

00537 NxsString::NxsQuotingRequirements NxsString::determine_quoting_requirements(const std::string & s)
00538     {
00539     NxsQuotingRequirements nrq = kNoQuotesNeededForNexus;
00540     for (std::string::const_iterator sIt = s.begin(); sIt != s.end(); ++sIt)
00541         {
00542         if (!isgraph(*sIt))
00543             {
00544             if (*sIt != ' ')
00545                 return kSingleQuotesNeededForNexus;
00546             nrq  = kUnderscoresSufficeForNexus;
00547             }
00548         else if (strchr("(){}\"-]/\\,;:=*`+<>", *sIt) != NULL)
00549             {
00550             // Get here if c is any NEXUS punctuation mark except left square bracket ([) or apostrophe (').
00551             // [ and ' never get returned as punctuation by NxsToken,
00552             // so we should never encounter them here.
00553             //
00554             return (s.length() > 1 ? kSingleQuotesNeededForNexus : kNoQuotesNeededForNexus);
00555             }
00556         else if (strchr("\'[_", *sIt) != NULL)
00557             {
00558             // Get here if c is either an apostrophe or left square bracket. Quotes are needed if one of these
00559             // characters is all there is to this string
00560             //
00561             return kSingleQuotesNeededForNexus;
00562             }
00563         }
00564     return nrq;
00565     }

00645 bool NxsString::IsADouble() const
00646     {
00647     const char  *str            = c_str();
00648     unsigned    i               = 0;
00649     bool        hadDecimalPt    = false;
00650     bool        hadExp          = false;
00651     bool        hadDigit        = false;
00652     bool        hadDigitInExp   = false;
00653 
00654     //  First char can be -
00655     //
00656     if (str[i]=='-' || str[i] == '+')
00657         i++;
00658 
00659     while (str[i])
00660         {
00661         if (isdigit(str[i]))
00662             {
00663             //  Digits are always OK
00664             //
00665             if (hadExp)
00666                 hadDigitInExp = true;
00667             else
00668                 hadDigit = true;
00669             }
00670         else if (str[i] == '.')
00671             {
00672             //  One decimal point is allowed and it must be before the exponent
00673             //
00674             if (hadExp || hadDecimalPt)
00675                 return false;
00676             hadDecimalPt = true;
00677             }
00678         else if (str[i] == 'e' || str[i] == 'E')
00679             {
00680             //  One e is allowed, but it must be after at least one digit
00681             //
00682             if (hadExp || !hadDigit)
00683                 return false;
00684             hadExp = true;
00685             }
00686         else if (str[i] == '-')
00687             {
00688             //  Another - is allowed if it is preceded by e
00689             //
00690             if (!hadExp || (str[i-1] != 'e' && str[i-1] != 'E') )
00691                 return false;
00692             }
00693         else
00694             return false;
00695         i++;
00696         }
00697 
00698     if (hadExp)
00699         {
00700         if (hadDigitInExp)
00701             return true;
00702         return false;
00703         }
00704 
00705     if (hadDigit)
00706         return true;
00707     return false;
00708     }
