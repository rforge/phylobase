00029 // splits a string by whitespace and push the graphical strings to the back of r.
00030 //  Leading and trailing whitespace is lost ( there will be no empty strings added
00031 //      to the list.
00032 void NxsString::split(const std::string &s, std::list<std::string> * r)
00033     {
00034     NCL_ASSERT(r);
00035     if (r == NULL)
00036         return;
00037     std::string current;
00038     for (std::string::const_iterator sIt = s.begin(); sIt != s.end(); ++sIt)
00039         {
00040         const char c = *sIt;
00041         if (isgraph(c))
00042             current.append(1, c);
00043         else if (!current.empty())
00044             {
00045             r->push_back(current);
00046             current.clear();
00047             }
00048         }
00049     if (!current.empty())
00050         r->push_back(current);
00051     }

00260 int NxsString::PrintF(
00261   const char *formatStr,    /* the printf-style format string */
00262   ...)                      /* other arguments referred to by the format string */
00263     {
00264     const int kInitialBufferSize = 256;
00265     char buf[kInitialBufferSize];
00266 
00267     // Create a pointer to the list of optional arguments
00268     //
00269     va_list argList;
00270 
00271     // Set arg_ptr to the first optional argument in argList. The
00272     // second argument (formatStr) is the last non-optional argument.
00273     //
00274     va_start(argList, formatStr);
00275 
00276     // If vsnprintf returns -1, means kInitialBufferSize was not large enough.
00277     // In this case, only kInitialBufferSize bytes are written.
00278     //
00279     int nAdded = vsnprintf(buf, kInitialBufferSize, formatStr, argList);
00280 
00281     // Reset the argument list pointer
00282     //
00283     va_end(argList);
00284 
00285     // Currently, if formatted string is too long to fit into the supplied buf,
00286     // just adding a terminating '\0' and returning the truncated string
00287     // Need to think of a better solution
00288     //
00289     if (nAdded  < 0 || nAdded >= kInitialBufferSize)
00290         buf[kInitialBufferSize - 1] = '\0';
00291 
00292     *this << buf;
00293 
00294     return nAdded;
00295     }

00300 bool NxsString::IsStdAbbreviation(
00301   const NxsString &s,   /* the string for which the stored string is potentially an abbreviation */
00302   bool respectCase)     /* if true, comparison will be case-sensitive */
00303   const
00304     {
00305     if (empty())
00306         return false;
00307 
00308     // s is the unabbreviated comparison string
00309     //
00310     const unsigned slen = static_cast<unsigned long>(s.size());
00311 
00312     // t is the stored string
00313     //
00314     const unsigned tlen = static_cast<unsigned long>(size());
00315 
00316     // t cannot be an abbreviation of s if it is longer than s
00317     //
00318     if (tlen > slen)
00319         return false;
00320 
00321     // Examine each character in t and return false (meaning "not an abbreviation")
00322     // if at any point the corresponding character in s is different
00323     //
00324     for (unsigned k = 0; k < tlen; k++)
00325         {
00326         if (respectCase)
00327             {
00328             if ((*this)[k] != s[k])
00329                 return false;
00330             }
00331         else if (toupper((*this)[k]) != toupper(s[k]))
00332             return false;
00333         }
00334 
00335     return true;
00336     }

00344 bool NxsString::IsCapAbbreviation(
00345   const NxsString &s)   /* the string for which the stored string is potentially an abbreviation */
00346   const
00347     {
00348     if (empty())
00349         return false;
00350 
00351     // s is the unabbreviated comparison string
00352     //
00353     const unsigned slen = static_cast<unsigned>(s.size());
00354 
00355     // t is the stored string
00356     //
00357     const unsigned tlen = static_cast<unsigned>(size());
00358 
00359     // If the stored string is longer than s then it cannot be an abbreviation of s
00360     //
00361     if (tlen > slen)
00362         return false;
00363 
00364     unsigned k = 0;
00365     for (; k < slen; k++)
00366         {
00367         if (isupper(s[k]))
00368             {
00369             // If still in the uppercase portion of s and we've run out of characters
00370             // in t, then t is not a valid abbrevation of s
00371             //
00372             if (k >= tlen)
00373                 return false;
00374 
00375             // If kth character in t is not equal to kth character in s, then
00376             // t is not an abbrevation of s
00377             //
00378             char tokenChar = (char)toupper((*this)[k]);
00379             if (tokenChar != s[k])
00380                 return false;
00381             }
00382         else if (!isalpha(s[k]))
00383             {
00384             // Get here if we are no longer in the upper case portion of s and
00385             // s[k] is not an alphabetic character. This section is necessary because
00386             // we are dealing with a section of s that is not alphabetical and thus
00387             // we cannot tell whether this should be part of the abbrevation or not
00388             // (i.e. we cannot tell if it is capitalized or not). In this case, we
00389             // pretend that we are still in the upper case portion of s and return
00390             // false if we have run out of characters in t (meaning that the abbreviation
00391             // was too short) or we find a mismatch.
00392             //
00393             if (k >= tlen)
00394                 return false;
00395 
00396             if ((*this)[k] != s[k])
00397                 return false;
00398             }
00399         else
00400             {
00401             // Get here if we are no longer in the upper case portion of s and
00402             // s[k] is an alphabetic character. Just break because we have determined
00403             // that t is in fact a valid abbreviation of s.
00404             //
00405             break;
00406             }
00407         }
00408 
00409     // Check the lower case portion of s and any corresponding characters in t for mismatches
00410     // Even though the abbreviation is valid up to this point, it will become invalid if
00411     // any mismatches are found beyond the upper case portion of s
00412     //
00413     for (; k < tlen; k++)
00414         {
00415         const char tokenChar = (char)toupper((*this)[k]);
00416         const char otherChar = (char)toupper(s[k]);
00417         if (tokenChar != otherChar)
00418             return false;
00419         }
00420 
00421     return true;
00422     }

00429 NxsString &NxsString::RightJustifyLong(
00430   long x,           /* long value to right justify */
00431   unsigned int w,   /* width of field */
00432   bool clear_first) /* if true, initialize string first to empty string */
00433     {
00434     bool x_negative = (x < 0L ? true : false);
00435     unsigned long xabs = (x_negative ? (-x) : x);
00436     unsigned num_spaces = w;
00437 
00438     // If w = 10 and x = 123, we need 7 blank spaces before x
00439     // log10(123) is 2.09, indicating that x is at least 10^2 = 100 but not
00440     // 10^3 = 1000, thus x requires at least 3 characters to display
00441     //
00442     unsigned x_width = (x == 0 ? 1 :1 + (int)log10((double)xabs));
00443     if (x_negative)
00444         x_width++;  // for the minus sign
00445 
00446     NCL_ASSERT(x_width <= num_spaces);
00447     num_spaces -= x_width;
00448 
00449     if (clear_first)
00450         erase();
00451 
00452     for (unsigned k = 0; k < num_spaces; k++)
00453         *this += ' ';
00454 
00455     if (x_negative)
00456         *this += '-';
00457 
00458     *this += xabs;
00459     return *this;
00460     }

00538 NxsString::NxsQuotingRequirements NxsString::determine_quoting_requirements(const std::string & s)
00539     {
00540     NxsQuotingRequirements nrq = kNoQuotesNeededForNexus;
00541     for (std::string::const_iterator sIt = s.begin(); sIt != s.end(); ++sIt)
00542         {
00543         if (!isgraph(*sIt))
00544             {
00545             if (*sIt != ' ')
00546                 return kSingleQuotesNeededForNexus;
00547             nrq  = kUnderscoresSufficeForNexus;
00548             }
00549         else if (strchr("(){}\"-]/\\,;:=*`+<>", *sIt) != NULL)
00550             {
00551             // Get here if c is any NEXUS punctuation mark except left square bracket ([) or apostrophe (').
00552             // [ and ' never get returned as punctuation by NxsToken,
00553             // so we should never encounter them here.
00554             //
00555             return (s.length() > 1 ? kSingleQuotesNeededForNexus : kNoQuotesNeededForNexus);
00556             }
00557         else if (strchr("\'[_", *sIt) != NULL)
00558             {
00559             // Get here if c is either an apostrophe or left square bracket. Quotes are needed if one of these
00560             // characters is all there is to this string
00561             //
00562             return kSingleQuotesNeededForNexus;
00563             }
00564         }
00565     return nrq;
00566     }

00646 bool NxsString::IsADouble() const
00647     {
00648     const char  *str            = c_str();
00649     unsigned    i               = 0;
00650     bool        hadDecimalPt    = false;
00651     bool        hadExp          = false;
00652     bool        hadDigit        = false;
00653     bool        hadDigitInExp   = false;
00654 
00655     //  First char can be -
00656     //
00657     if (str[i]=='-' || str[i] == '+')
00658         i++;
00659 
00660     while (str[i])
00661         {
00662         if (isdigit(str[i]))
00663             {
00664             //  Digits are always OK
00665             //
00666             if (hadExp)
00667                 hadDigitInExp = true;
00668             else
00669                 hadDigit = true;
00670             }
00671         else if (str[i] == '.')
00672             {
00673             //  One decimal point is allowed and it must be before the exponent
00674             //
00675             if (hadExp || hadDecimalPt)
00676                 return false;
00677             hadDecimalPt = true;
00678             }
00679         else if (str[i] == 'e' || str[i] == 'E')
00680             {
00681             //  One e is allowed, but it must be after at least one digit
00682             //
00683             if (hadExp || !hadDigit)
00684                 return false;
00685             hadExp = true;
00686             }
00687         else if (str[i] == '-')
00688             {
00689             //  Another - is allowed if it is preceded by e
00690             //
00691             if (!hadExp || (str[i-1] != 'e' && str[i-1] != 'E') )
00692                 return false;
00693             }
00694         else
00695             return false;
00696         i++;
00697         }
00698 
00699     if (hadExp)
00700         {
00701         if (hadDigitInExp)
00702             return true;
00703         return false;
00704         }
00705 
00706     if (hadDigit)
00707         return true;
00708     return false;
00709     }
