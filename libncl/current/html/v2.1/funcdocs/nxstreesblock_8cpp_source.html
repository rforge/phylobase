00521     {

00523         return comment;
00524     size_t colonPos = comment.find(':', 5);
00525     if (colonPos == string::npos)
00526         return comment.substr(5, string::npos);
00527     for (;;)
00528         {
00529         size_t eqPos = comment.find('=', colonPos);
00530         if (eqPos == string::npos || (eqPos <= (colonPos + 1)))
00531             return comment.substr(colonPos, string::npos);
00532         std::string key = comment.substr(colonPos + 1, eqPos - 1 - colonPos);
00533         colonPos = comment.find(':', eqPos + 1);
00534         if (colonPos == eqPos + 1)
00535             {
00536             if (infoMap)
00537                 (*infoMap)[key] = string();
00538             }
00539         else if (colonPos == string::npos)
00540             {
00541             std::string lastVal = comment.substr(eqPos + 1);
00542             if (infoMap)
00543                 (*infoMap)[key] = lastVal;
00544             return std::string();
00545             }
00546         else
00547             {
00548             std::string value = comment.substr(eqPos + 1, colonPos - eqPos - 1);
00549             if (infoMap)

00683             }

01241                         throw NxsException(errormsg, token);
01242                     /* if we did not throw an excection, then we are in relaxed parsing mode.
01243                         We'll add the implied ,
01244                     */
01245                     if (!someMissingEdgeLens && (prevToken == NXS_TREE_CLOSE_PARENS_TOKEN || prevToken == NXS_TREE_CLADE_NAME_TOKEN))
01246                         someMissingEdgeLens = true;
01247                     nchildren.top() += 1;
01248                     newickStream << ',';
01249                     prevToken = NXS_TREE_COMMA_TOKEN;
01250                     }
01251                 else if (prevToken == NXS_TREE_COLON_TOKEN)
01252                     throw NxsException("Expecting a branch length after a : but found (", token);
01253                 nchildren.top() += 1;
01254                 nchildren.push(0);
01255                 newickStream << '(';
01256                 prevToken = NXS_TREE_OPEN_PARENS_TOKEN;
01257                 handled = true;
01258                 }
01259             else if (t[0] == ')')
01260                 {
01261                 if (nchildren.empty())
01262                     throw NxsException("End of tree description.  Expected ; but found )", token);
01263                 if (prevToken == NXS_TREE_OPEN_PARENS_TOKEN || prevToken == NXS_TREE_COMMA_TOKEN)
01264                     throw NxsException("Expecting a clade description before the subtree's closing )\n \"()\" and \",)\" are prohibited.", token);
01265                 if (prevToken == NXS_TREE_COLON_TOKEN)
01266                     throw NxsException("Expecting a branch length after a : but found (", token);
01267                 if (!someMissingEdgeLens && (prevToken == NXS_TREE_CLOSE_PARENS_TOKEN || prevToken == NXS_TREE_CLADE_NAME_TOKEN))
01268                     someMissingEdgeLens = true;
01269                 if (nchildren.top() == 1)
01270                     hasDegTwoNodes = true;
01271                 else if ((!hasPolytomies) && (nchildren.top() > 2))
01272                     {
01273                     if (rooted)
01274                         {
01275                         if (nchildren.top() > 2)
01276                             hasPolytomies = true;
01277                         }
01278                     else if (nchildren.top() > 3 || nchildren.size() > 1) /* three children are allowed not considered a polytomy */
01279                         hasPolytomies = true;
01280                     }
01281                 nchildren.pop();
01282                 newickStream << ')';
01283                 prevToken = NXS_TREE_CLOSE_PARENS_TOKEN;
01284                 handled = true;
01285                 }
01286             else if (t[0] == ':')
01287                 {
01288                 if (prevToken != NXS_TREE_CLOSE_PARENS_TOKEN && prevToken != NXS_TREE_CLADE_NAME_TOKEN)
01289                     throw NxsException("Found a : separator for a subtree at an inappropriate location. A colon is only permitted after a clade name or )-symbol.", token);
01290                 if (taxsetRead && prevToken == NXS_TREE_CLADE_NAME_TOKEN)
01291                     throw NxsException("Found a : separator after a taxset name. Branch lengths cannot be assigned to multi-taxon taxsets.", token);
01292                 newickStream << ':';
01293                 prevToken = NXS_TREE_COLON_TOKEN;
01294                 handled = true;
01295                 token.SetLabileFlagBit(NxsToken::hyphenNotPunctuation); // this allows us to deal with sci. not. in branchlengths (and negative branch lengths).
01296                 }
01297             else if (t[0] == ',')
01298                 {
01299                 if (prevToken == NXS_TREE_OPEN_PARENS_TOKEN)
01300                     throw NxsException("Found a empty subclade found. The combination \"(,\" is prohibited.", token);
01301                 if (prevToken == NXS_TREE_COMMA_TOKEN)
01302                     throw NxsException("Found a empty subclade found. The combination \",,\" is prohibited.", token);
01303                 if (prevToken == NXS_TREE_COLON_TOKEN)
01304                     throw NxsException("Found a , when a branch length was expected found. The combination \":,\" is prohibited.", token);
01305                 if (!someMissingEdgeLens && (prevToken == NXS_TREE_CLOSE_PARENS_TOKEN || prevToken == NXS_TREE_CLADE_NAME_TOKEN))
01306                     someMissingEdgeLens = true;
01307                 nchildren.top() += 1;
01308                 newickStream << ',';
01309                 prevToken = NXS_TREE_COMMA_TOKEN;
01310                 handled = true;
01311                 }
01312             }
01313         if (!handled)
01314             {
01315             if (prevToken == NXS_TREE_COLON_TOKEN)
01316                 {
01317                 bool handledLength = false;
01318                 if (!someRealEdgeLens)
01319                     {
01320                     if (NxsString::to_long(t, &lastIntEdgeLen))
01321                         {
01322                         handledLength = true;
01323                         if (lastIntEdgeLen < minIntEdgeLen)
01324                             minIntEdgeLen = lastIntEdgeLen;
01325                         }
01326                     }
01327                 if (!handledLength)
01328                     {
01329                     if (!NxsString::to_double(t, &lastFltEdgeLen))
01330                         {
01331                         errormsg << "Expecting a number as a branch length. Found " << tstr;
01332                         throw NxsException(errormsg, token);
01333                         }
01334                     someRealEdgeLens = true;
01335                     if (lastFltEdgeLen < minDblEdgeLen)
01336                         minDblEdgeLen = lastFltEdgeLen;
01337                     }
01338                 newickStream << tstr;
01339                 someHaveEdgeLens = true;
01340                 prevToken = NXS_TREE_BRLEN_TOKEN;
01341                 }
01342             else
01343                 {
01344                 if (prevToken == NXS_TREE_BRLEN_TOKEN || prevToken == NXS_TREE_CLADE_NAME_TOKEN)
01345                     {
01346                     errormsg << "Found a name " << tstr << " which should be preceded by a ( or a ,";
01347                     throw NxsException(errormsg, token);
01348                     }
01349                 taxsetRead = false;
01350                 NxsString ucl(t);
01351                 if (!respectCase)
01352                     ucl.ToUpper();

01354                 std::map<std::string, unsigned>::const_iterator tt = capNameToInd.find(ucl);
01355                 unsigned ind = (tt == capNameToInd.end() ? UINT_MAX : tt->second);

01357                     {
01358                     errormsg << "Taxon number " << ind + 1 << " (coded by the token " << tstr << ") has already been encountered in this tree. Duplication of taxa in a tree is prohibited.";
01359                     throw NxsException(errormsg, token);
01360                     }
01361                 if (prevToken == NXS_TREE_CLOSE_PARENS_TOKEN)
01362                     {
01363                     hasInternalLabels = true;
01364                     if (ind == UINT_MAX)
01365                         {
01366                         hasInternalLabelsNotInTaxa = true;
01367                         toAppend += tstr;
01368                         }
01369                     else
01370                         {
01371                         hasInternalLabelsInTaxa = true;
01372                         taxaEncountered.insert(ind);
01373                         toAppend += (1 + ind);
01374                         }
01375                     }
01376                 else
01377                     {
01378                     if (ind == UINT_MAX)
01379                         {
01380                         std::set<unsigned> csinds;
01381                         unsigned nadded = taxa->GetIndexSet(tstr, &csinds);
01382                         if (nadded == 0)
01383                             {
01384                             if (!allowNewTaxa)
01385                                 {
01386                                 errormsg << "Expecting a Taxon label after a \"" << (prevToken == NXS_TREE_OPEN_PARENS_TOKEN ? '(' : ',') << "\" character. Found \"" << tstr << "\" but this is not a recognized taxon label.";
01387                                 throw NxsException(errormsg, token);
01388                                 }
01389                             std::string tasstring(tstr.c_str());
01390                             unsigned valueInd = taxa->AppendNewLabel(tasstring);
01391                             NxsString numV;
01392                             numV += (valueInd+1);
01393                             if (capNameToInd.find(numV) == capNameToInd.end())
01394                                 capNameToInd[numV] = valueInd;
01395                             if (!respectCase)
01396                                 NxsString::to_upper(tasstring);
01397                             capNameToInd[tasstring] = valueInd;
01398                             toAppend += (1 + valueInd);
01399                             }
01400                         else
01401                             {
01402                             bool firstTaxonAdded = true;
01403                             for (std::set<unsigned>::const_iterator cit = csinds.begin(); cit != csinds.end(); ++cit)
01404                                 {
01405                                 if (taxaEncountered.find(*cit) != taxaEncountered.end())
01406                                     {
01407                                     errormsg << "Taxon number " << *cit + 1 << " (one of the members of the taxset " << tstr << ") has already been encountered in this tree. Duplication of taxa in a tree is prohibited.";
01408                                     throw NxsException(errormsg, token);
01409                                     }
01410                                 taxaEncountered.insert(*cit);
01411                                 if (!firstTaxonAdded)
01412                                     toAppend.append(1, ',');
01413                                 toAppend += (1 + *cit);
01414                                 firstTaxonAdded = false;
01415                                 }
01416                             if (nadded > 1)
01417                                 {
01418                                 taxsetRead = true;
01419                                 someMissingEdgeLens = true;
01420                                 }
01421                             }
01422                         }
01423                     else
01424                         {
01425                         taxaEncountered.insert(ind);
01426                         toAppend += (1 + ind);
01427                         }
01428                     }
01429                 newickStream << toAppend;
01430                 prevToken = NXS_TREE_CLADE_NAME_TOKEN;
01431                 }
01432             }
01433         token.GetNextToken();
01434         }
01435     td.flags |= NxsFullTreeDescription::NXS_TREE_PROCESSED;
01436     if (someHaveEdgeLens)
01437         {
01438         flags |= NxsFullTreeDescription::NXS_HAS_SOME_EDGE_LENGTHS_BIT;
01439         if (someRealEdgeLens)
01440             {
01441             flags &= ~(NxsFullTreeDescription::NXS_INT_EDGE_LENGTHS_BIT);
01442             td.minDblEdgeLen = minDblEdgeLen;
01443             }
01444         else
01445             {
01446             flags |= NxsFullTreeDescription::NXS_INT_EDGE_LENGTHS_BIT;
01447             td.minIntEdgeLen = minIntEdgeLen;
01448             }
01449         }
01450     td.newick = newickStream.str();
01451     if (someMissingEdgeLens)
01452         flags |= NxsFullTreeDescription::NXS_MISSING_SOME_EDGE_LENGTHS_BIT;
01453     if (hasPolytomies)
01454         flags |= NxsFullTreeDescription::NXS_HAS_POLYTOMY_BIT;
01455     if (hasDegTwoNodes)
01456         flags |= NxsFullTreeDescription::NXS_HAS_DEG_TWO_NODES_BIT;
01457     if (hasInternalLabels)
01458         {
01459         flags |= NxsFullTreeDescription::NXS_HAS_INTERNAL_NAMES_BIT;
01460         if (hasInternalLabelsNotInTaxa)
01461             flags |= NxsFullTreeDescription::NXS_HAS_NEW_INTERNAL_NAMES_BIT;
01462         if (hasInternalLabelsInTaxa)
01463             flags |= NxsFullTreeDescription::NXS_KNOWN_INTERNAL_NAMES_BIT;
01464         }
01465     if (NHXComments)
01466         flags |= NxsFullTreeDescription::NXS_HAS_NHX_BIT;
01467     if (taxaEncountered.size() == taxa->GetMaxIndex() + 1)
01468         flags |= NxsFullTreeDescription::NXS_HAS_ALL_TAXA_BIT;
01469     }

01483 void NxsTreesBlock::HandleTreeCommand(NxsToken &token, bool rooted)
01484     {
01485     NCL_ASSERT(taxa);
01486     token.GetNextToken();
01487     if (token.Equals("*"))
01488         {
01489         defaultTreeInd = (unsigned)trees.size();
01490         token.GetNextToken();
01491         }
01492     NxsString treeName = token.GetToken();
01493     DemandEquals(token, "after tree name in TREE command");
01494     file_pos fp = 0;
01495     int fline = token.GetFileLine();
01496     int fcol = token.GetFileColumn();
01497     fp = token.GetFilePosition();
01498     try {
01499         // This should be either a tree description or a command comment specifying
01500         // whether this tree is to be rooted ([&R]) or unrooted ([&U]).
01501         //
01502         token.SetLabileFlagBit(NxsToken::saveCommandComments);
01503         token.SetLabileFlagBit(NxsToken::parentheticalToken);
01504         token.GetNextToken();
01505         NxsString s = token.GetToken();
01506         if (!s.empty() && s[0] == '&')
01507             {
01508             if (s[1] == 'R' || s[1] == 'r')
01509                 rooted = true;
01510             else if (s[1] == 'U' || s[1] == 'u')
01511                 rooted = false;
01512             else
01513                 {
01514                 errormsg << "[" << token.GetToken() << "] is not a valid command comment in a TREE command";
01515                 throw NxsException(errormsg, token.GetFilePosition(), token.GetFileLine(), token.GetFileColumn());
01516                 }
01517             // now grab the tree description
01518             token.SetLabileFlagBit(NxsToken::parentheticalToken);
01519             token.GetNextToken();
01520             s = token.GetToken();
01521             }
01522         if (!s.empty() && s[0] != '(')
01523             {
01524             errormsg << "Expecting command comment or tree description in TREE (or UTREE) command, but found " << token.GetToken() << " instead";
01525             throw NxsException(errormsg);
01526             }
01527         }
01528     catch (NxsX_UnexpectedEOF &)
01529         {
01530         errormsg << "Unexpected end of file in tree description.\n";
01531         errormsg << "This probably indicates that the parentheses in the newick description are not balanced, and one or more closing parentheses are needed.";
01532         throw NxsException(errormsg, fp, fline, fcol);
01533         }
01534     std::string mt;
01535     int f = (rooted ? NxsFullTreeDescription::NXS_IS_ROOTED_BIT : 0);
01536     trees.push_back(NxsFullTreeDescription(mt, treeName, f));
01537     NxsFullTreeDescription & td = trees[trees.size() -1];
01538     ReadTreeFromOpenParensToken(td, token);
01539     }

01541 void NxsTreesBlock::ReadTreeFromOpenParensToken(NxsFullTreeDescription &td, NxsToken & token)
01542     {
01543     file_pos fp = 0;
01544     int fline = token.GetFileLine();
01545     int fcol = token.GetFileColumn();
01546     ostringstream newickStream;
01547     newickStream << token.GetTokenReference();
01548     token.GetNextToken();
01549     const std::vector<NxsComment> & ecs = token.GetEmbeddedComments();
01550     for (std::vector<NxsComment>::const_iterator ecsIt = ecs.begin(); ecsIt != ecs.end(); ++ecsIt)
01551         ecsIt->WriteAsNexus(newickStream);
01552     while (!token.Equals(";"))
01553         {
01554         if (token.Equals("(") || token.Equals(")") || token.Equals(","))
01555             GenerateUnexpectedTokenNxsException(token, "root taxon information");
01556         newickStream << NxsString::GetEscaped(token.GetTokenReference());
01557         token.GetNextToken();
01558         const std::vector<NxsComment> & iecs = token.GetEmbeddedComments();
01559         for (std::vector<NxsComment>::const_iterator iecsIt = iecs.begin(); iecsIt != iecs.end(); ++iecsIt)
01560             iecsIt->WriteAsNexus(newickStream);
01561         }
01562     td.newick = newickStream.str();
01563     if (processAllTreesDuringParse)
01564         {
01565         try
01566             {
01567             ProcessTree(td);
01568             if (this->processedTreeValidationFunction)
01569                 {
01570                 if (!this->processedTreeValidationFunction(td, this->ptvArg, this))
01571                     trees.pop_back();
01572                 }
01573             }
01574         catch (NxsException &x)
01575             {
01576             x.pos += fp;
01577             x.line += fline - 1; /*both tokenizers start at 1 instead of zero, so we need to decrement the line */
01578             x.col += fcol;
01579             throw x;
01580             }
01581         }

01588 void NxsTreesBlock::Read(
01589   NxsToken &token)  /* the token used to read from `in' */
01590     {
01591     isEmpty = false;
01592     isUserSupplied = true;
01593     DemandEndSemicolon(token, "BEGIN TREES");
01594     //AssureTaxaBlock(createImpliedBlock, token, "BEGIN TREES");
01595     bool readTranslate = false;
01596     bool readTree = false;
01597     errormsg.clear();
01598     constructingTaxaBlock = false;
01599     newtaxa = false;
01600     capNameToInd.clear();
01601     unsigned numSigInts = NxsReader::getNumSignalIntsCaught();
01602     const bool checkingSignals = NxsReader::getNCLCatchesSignals();
01603 
01604     for (;;)
01605         {
01606         token.GetNextToken();
01607         if (checkingSignals && NxsReader::getNumSignalIntsCaught() != numSigInts)
01608             {
01609             throw NxsSignalCanceledParseException("Reading TREES Block");
01610             }
01611         NxsBlock::NxsCommandResult res = HandleBasicBlockCommands(token);
01612         if (res == NxsBlock::NxsCommandResult(STOP_PARSING_BLOCK))
01613             {
01614             if (constructingTaxaBlock)
01615                 {
01616                 if (taxa && taxa->GetNTax() > 0)
01617                     newtaxa = true;
01618                 constructingTaxaBlock = false; /* we don't allow the construction of taxa blocks over repeated readings or after the block has been read */
01619                 }
01620             return;
01621             }
01622         if (res != NxsBlock::NxsCommandResult(HANDLED_COMMAND))
01623             {
01624             if (token.Equals("TRANSLATE"))
01625                 {

01628                 if (readTranslate)
01629                     {
01630                     WarnDangerousContent("Only one TRANSLATE command may be read in a TREES block", token);
01631                     capNameToInd.clear();
01632                     }
01633                 readTranslate = true;
01634                 ConstructDefaultTranslateTable(token, "TRANSLATE");
01635                 HandleTranslateCommand(token);
01636                 }
01637             else
01638                 {
01639                 bool utreeCmd = token.Equals("UTREE");
01640                 bool treeCmd = token.Equals("TREE");
01641                 if (utreeCmd || treeCmd)
01642                     {
01643                     if (!readTranslate && ! readTree)
01644                         ConstructDefaultTranslateTable(token, token.GetTokenReference().c_str());
01645                     readTree = true;
01646                     HandleTreeCommand(token, treeCmd);
01647                     }
01648                 else
01649                     SkipCommand(token);
01650                 }
01651             }
01652         }

01659 NxsString NxsTreesBlock::GetTranslatedTreeDescription(
01660   unsigned i)   /* the index of the tree for which the description is to be returned */
01661     {
01662     NCL_ASSERT(i < trees.size());
01663     NCL_ASSERT(taxa);
01664     NxsFullTreeDescription & ftd = trees.at(i);
01665     ProcessTree(ftd);
01666     std::string incomingNewick = ftd.newick;
01667     incomingNewick.append(1, ';');
01668     istringstream newickstream(incomingNewick);
01669     NxsToken token(newickstream);
01670     token.GetNextToken();
01671     if (!token.Equals("("))
01672         {
01673         errormsg << "Expecting a ( to start the tree description, but found " << token.GetTokenReference();
01674         throw NxsException(errormsg, token);
01675         }
01676     int prevToken = NXS_TREE_OPEN_PARENS_TOKEN;
01677     long taxIndLong;
01678     const unsigned ntax = taxa->GetNTaxTotal();
01679     ostringstream translated;
01680     for (;;)
01681         {
01682         const std::vector<NxsComment> & ecs = token.GetEmbeddedComments();
01683         for (std::vector<NxsComment>::const_iterator ecsIt = ecs.begin(); ecsIt != ecs.end(); ++ecsIt)
01684             ecsIt->WriteAsNexus(translated);
01685         if (token.Equals(";"))
01686             break;
01687         const NxsString & t = token.GetTokenReference();
01688         bool handled;
01689         handled = false;
01690         if (t.length() == 1)
01691             {
01692             if (t[0] == '(')
01693                 {
01694                 translated <<  '(';
01695                 prevToken = NXS_TREE_OPEN_PARENS_TOKEN;
01696                 handled = true;
01697                 }
01698             else if (t[0] == ')')
01699                 {
01700                 translated << ')';
01701                 prevToken = NXS_TREE_CLOSE_PARENS_TOKEN;
01702                 handled = true;
01703                 }
01704             else if (t[0] == ':')
01705                 {
01706                 translated << ':';
01707                 prevToken = NXS_TREE_COLON_TOKEN;
01708                 handled = true;
01709                 token.SetLabileFlagBit(NxsToken::hyphenNotPunctuation); // this allows us to deal with sci. not. in branchlengths (and negative branch lengths).
01710                 }
01711             else if (t[0] == ',')
01712                 {
01713                 translated << ',';
01714                 prevToken = NXS_TREE_COMMA_TOKEN;
01715                 handled = true;
01716                 }
01717             }
01718         if (!handled)
01719             {
01720             if (prevToken == NXS_TREE_COLON_TOKEN)
01721                 {
01722                 translated << t;
01723                 prevToken = NXS_TREE_BRLEN_TOKEN;
01724                 }
01725             else
01726                 {
01727                 if (NxsString::to_long(t.c_str(), &taxIndLong) && taxIndLong <= (long) ntax && taxIndLong > 0)
01728                     translated << NxsString::GetEscaped(taxa->GetTaxonLabel((unsigned) taxIndLong - 1));
01729                 else if (prevToken == NXS_TREE_CLOSE_PARENS_TOKEN)
01730                     translated << t;
01731                 else
01732                     {
01733                     errormsg << "Expecting a taxon index in a tree description, but found " << t;
01734                     throw NxsException(errormsg, token);
01735                     }
01736                 }
01737             }
01738         token.GetNextToken();
01739         }
01740     return NxsString(translated.str().c_str());
01741     }
