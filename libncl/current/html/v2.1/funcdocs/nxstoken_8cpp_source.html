00247     {
00248     //
00249     //  Why this was changed:  calls to tellg seem slow and unnecessary - we're storing filepos in terms of the
00250     //  number of times we call sbumpc().

00255     AdvanceToNextCharInStream();

00260             return '\0';

00264         {
00265         fileLine++;
00266         fileColumn = 1L;
00267         atEOL = true;
00268         return '\n';

00271         fileColumn += 4 - ((fileColumn - 1)%4); //@assumes that tab will be 4 in the editor we use

00523 bool NxsToken::GetComment()
00524     {
00525     // Set comment level to 1 initially.  Every ']' encountered reduces
00526     // level by one, so that we know we can stop when level becomes 0.
00527     //
00528     NxsString currentComment;
00529     bool command = false;
00530 
00531     bool formerEOFAllowed = eofAllowed;
00532     eofAllowed = false;
00533     try
00534         {
00535         char ch = GetNextChar();
00536         // See if first character is the output comment symbol ('!')
00537         // or command comment symbol (&)
00538         //
00539         int printing = 0;
00540         if (ch == '!')
00541             printing = 1;
00542         else if (ch == '&' && (labileFlags & saveCommandComments) && token.empty())
00543             command = true;
00544         currentComment.push_back(ch);
00545         if (ch != ']')
00546             {
00547             int level = 1;
00548             for(;;)
00549                 {
00550                 ch = GetNextChar();
00551                 if (ch == ']')
00552                     {
00553                     level--;
00554                     if (level == 0)
00555                         break;
00556                     }
00557                 else if (ch == '[')
00558                     level++;
00559                 currentComment.push_back(ch);
00560                 }
00561 
00562             if (printing)
00563                 {
00564                 // Allow output comment to be printed or displayed in most appropriate
00565                 // manner for target operating system
00566                 //
00567                 NxsString foroutput(currentComment.c_str() + 1);
00568                 comment = foroutput;
00569                 OutputComment(foroutput);
00570                 }
00571             if (command)
00572                 token = currentComment;
00573             else
00574                 embeddedComments.push_back(NxsComment(currentComment, GetFileLine(), GetFileColumn()));
00575             }
00576         }
00577     catch (NxsX_UnexpectedEOF & x)
00578         {
00579         x.msg << " (end-of-file inside comment)";
00580         eofAllowed = formerEOFAllowed;
00581         throw x;
00582         }
00583     eofAllowed = formerEOFAllowed ;
00584     return command;
00585     }

00727 bool NxsToken::Abbreviation(
00728   NxsString s)  /* the comparison string */
00729     {
00730     int k;
00731     int slen = (int)s.size();
00732     int tlen = (int)token.size();
00733     char tokenChar, otherChar;
00734 
00735     // The variable mlen refers to the "mandatory" portion
00736     // that is the upper-case portion of s
00737     //
00738     int mlen;
00739     for (mlen = 0; mlen < slen; mlen++)
00740         {
00741         if (!isupper(s[mlen]))
00742             break;
00743         }
00744 
00745     // User must have typed at least mlen characters in
00746     // for there to even be a chance at a match
00747     //
00748     if (tlen < mlen)
00749         return false;
00750 
00751     // If user typed in more characters than are contained in s,
00752     // then there must be a mismatch
00753     //
00754     if (tlen > slen)
00755         return false;
00756 
00757     // Check the mandatory portion for mismatches
00758     //
00759     for (k = 0; k < mlen; k++)
00760         {
00761         tokenChar = (char)toupper( token[k]);
00762         otherChar = s[k];
00763         if (tokenChar != otherChar)
00764             return false;
00765         }
00766 
00767     // Check the auxiliary portion for mismatches (if necessary)
00768     //
00769     for (k = mlen; k < tlen; k++)
00770         {
00771         tokenChar = (char)toupper( token[k]);
00772         otherChar = (char)toupper( s[k]);
00773         if (tokenChar != otherChar)
00774             return false;
00775         }
00776 
00777     return true;
00778     }
